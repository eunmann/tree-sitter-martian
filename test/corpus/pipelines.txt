pipeline P(
    in  fastq reads,
    out bam   aligned,
)
{
    call local preflight ALIGN(
        reads = self.reads,
    )
    call BUILD as BUILDER(
        x = ALIGN.aligned,
    ) using (
        volatile = true,
        disabled = self.skip,
    )
    return (
        aligned = ALIGN.aligned,
    )
    retain (
        ALIGN.aligned,
    )
}

================================================================
pipeline with only return
================================================================

pipeline EMPTY(
    out int x,
)
{
    return (
        x = 0,
    )
}

================================================================
map call with split bindings and wildcard
================================================================

pipeline P(
    out int x,
)
{
    map call WORKER(
        item = split [1, 2, 3],
        *    = self,
    )
    return (
        x = 1,
    )
}
